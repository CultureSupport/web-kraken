    <script>
        // --- Tone.js & Looper Logic ---

        // 1. Setup global state and components
        let synth = null;
        let recordedNotes = [];
        let isRecording = false;
        let looperSequence = null;
        let loopLength = '4m'; // 4 measures is the default loop length
        const noteMap = {
            'a': { note: 'C4', elementId: 'key-A' },
            's': { note: 'D4', elementId: 'key-S' },
            'd': { note: 'E4', elementId: 'key-D' },
            'f': { note: 'F4', elementId: 'key-F' },
            'g': { note: 'G4', elementId: 'key-G' },
            'h': { note: 'A4', elementId: 'key-H' },
            'j': { note: 'B4', elementId: 'key-J' },
            'k': { note: 'C5', elementId: 'key-K' }
        };

        // 2. UI Elements
        const statusEl = document.getElementById('loop-status');
        const countEl = document.getElementById('note-count');
        const recordBtn = document.getElementById('record-button');
        const startBtn = document.getElementById('start-button');
        const stopBtn = document.getElementById('stop-button');
        const allKeys = Object.keys(noteMap);

        // Function to update the UI status display
        const updateStatus = (status, color) => {
            Tone.Draw.schedule(() => {
                statusEl.textContent = status;
                statusEl.className = `text-sm font-semibold uppercase ${color}`;
            }, Tone.now());
        };

        const updateNoteCount = (count) => {
            Tone.Draw.schedule(() => {
                countEl.textContent = `${count} Notes`;
            }, Tone.now());
        };

        // 3. Audio Initialization
        const initializeAudio = async () => {
            if (Tone.context.state !== 'running') {
                await Tone.start();
                updateStatus('Ready (Audio Started)', 'text-green-400');
            }

            // Create a simple, clean synth voice
            synth = new Tone.PolySynth(Tone.Synth, {
                oscillator: { type: 'triangle' },
                envelope: {
                    attack: 0.005,
                    decay: 0.1,
                    sustain: 0.3,
                    release: 1
                }
            }).toDestination();

            // Setup Tone Transport
            Tone.Transport.bpm.value = 120;
            Tone.Transport.loop = true;
            Tone.Transport.loopEnd = loopLength;
            document.getElementById('tempo-display').textContent = `${Tone.Transport.bpm.value} BPM`;
        };

        // 4. Looping Control
        const createLooper = () => {
            // Clear any existing sequence
            if (looperSequence) {
                looperSequence.stop();
                looperSequence.dispose();
            }

            // Create a new sequence from the recorded notes
            looperSequence = new Tone.Sequence((time, noteEvent) => {
                // Play the note with its duration
                synth.triggerAttackRelease(noteEvent.note, noteEvent.duration, time);
            }, recordedNotes, 0.5).start(0);

            // The '0.5' above is the subdivision. If notes were recorded by time, use a small subdivision.
            // Since we stored the time offset, we use '0.5' and rely on the noteEvents time property.
        };

        // 5. Recording Functions
        const startRecording = () => {
            // Ensure audio is initialized
            if (!synth) initializeAudio();

            // Clear previous loop
            if (looperSequence) {
                looperSequence.stop();
                looperSequence.dispose();
                looperSequence = null;
            }

            // Clear recorded notes
            recordedNotes = [];
            updateNoteCount(0);

            isRecording = true;
            Tone.Transport.stop();
            Tone.Transport.seconds = 0; // Reset transport position
            Tone.Transport.start();

            recordBtn.classList.remove('bg-red-600', 'hover:bg-red-700');
            recordBtn.classList.add('bg-gray-400', 'animate-pulse');
            updateStatus('Recording...', 'text-red-500');
        };

        const stopRecording = () => {
            isRecording = false;
            Tone.Transport.stop();

            // Restore button appearance
            recordBtn.classList.remove('bg-gray-400', 'animate-pulse');
            recordBtn.classList.add('bg-red-600', 'hover:bg-red-700');

            if (recordedNotes.length > 0) {
                // Sort notes by time for correct playback
                recordedNotes.sort((a, b) => a.time - b.time);
                updateStatus(`Recorded ${recordedNotes.length} Notes`, 'text-blue-400');
                updateNoteCount(recordedNotes.length);

                // Now the user can press START to play the loop
            } else {
                updateStatus('Recording Failed. No notes captured.', 'text-yellow-400');
            }
        };

        // 6. Note Playback and Recording Logic
        const playNote = (key) => {
            if (!synth) initializeAudio(); // Initialize on first key press

            const { note, elementId } = noteMap[key];

            // 6a. Visual feedback
            const keyEl = document.getElementById(elementId);
            if (keyEl && !keyEl.classList.contains('key-active')) {
                keyEl.classList.add('key-active');
                setTimeout(() => keyEl.classList.remove('key-active'), 100);
            }

            // 6b. Trigger audio
            synth.triggerAttackRelease(note, '8n');

            // 6c. Record note if recording is active
            if (isRecording) {
                const time = Tone.Transport.seconds;
                recordedNotes.push({
                    time: time,
                    note: note,
                    duration: '8n' // Fixed duration for simplicity
                });
                updateNoteCount(recordedNotes.length);
            }
        };

        // 7. Event Listeners
        const setupListeners = () => {
            // Global key press listener
            document.addEventListener('keydown', (e) => {
                const key = e.key.toLowerCase();
                if (noteMap[key] && !e.repeat) {
                    playNote(key);
                }
            });

            // Click listeners for the virtual keyboard
            document.getElementById('keyboard-pad').addEventListener('click', (e) => {
                const keyId = e.target.closest('.key-button')?.id;
                if (keyId) {
                    const keyChar = keyId.replace('key-', '').toLowerCase();
                    if (noteMap[keyChar]) {
                        playNote(keyChar);
                    }
                }
            });


            // Start/Play Button
            startBtn.addEventListener('click', () => {
                if (recordedNotes.length === 0) {
                    updateStatus('Error: No notes recorded to play!', 'text-red-600');
                    return;
                }
                // Stop any ongoing recording
                if (isRecording) stopRecording();

                // Create looper if it doesn't exist (or was disposed)
                createLooper();

                // Start transport only if it's not already running
                if (Tone.Transport.state !== 'started') {
                    Tone.Transport.start();
                    updateStatus('Playing Loop', 'text-green-500');
                }
            });

            // Record Button
            recordBtn.addEventListener('click', () => {
                if (isRecording) {
                    stopRecording();
                } else {
                    startRecording();
                }
            });

            // Stop Button
            stopBtn.addEventListener('click', () => {
                if (looperSequence) looperSequence.stop();
                if (isRecording) stopRecording();
                Tone.Transport.stop();
                updateStatus('Stopped', 'text-gray-400');
            });

            // Initialize audio on the first interaction (required by modern browsers)
            document.addEventListener('click', initializeAudio, { once: true });
        };

        // Initialize the app when the window loads
        window.onload = setupListeners;

    </script>
